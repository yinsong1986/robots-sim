# Isaac Sim + 3DGS hybrid-render example — runtime deps.
#
# The heavy backend (Isaac Sim itself) is NOT pip-installable; install it
# separately (RTX GPU + Omniverse / Isaac Lab / NGC docker) and then:
#   pip install 'strands-robots-sim[isaac]'
#
# This file lists only the light extras the example's render/composite path
# needs on top of a working Isaac Sim + strands-robots-sim[isaac].

# Image IO for saving composited PNGs (falls back to numpy .npy if absent).
pillow>=10.0

# Video assembly for the multi-frame / --wave clip output (MP4 via libx264,
# or GIF). Matches examples/mujoco_gs's imageio encode. PNG stills still work
# without these; only the assembled clip needs them.
imageio>=2.19
imageio-ffmpeg>=0.4  # ffmpeg backend for MP4 (libx264)

# numpy is already pulled in by strands-robots-sim; pinned loosely here for
# the standalone-install case.
numpy>=1.24

# Optional: real 3D Gaussian Splatting backgrounds (the digital-twin path).
#
# gsplat needs CUDA. A plain `pip install gsplat` builds the CUDA kernels
# from source at first use via nvcc -- which the Isaac Sim container does
# NOT ship (it has the CUDA *runtime* for RTX, but no CUDA *toolkit*). In
# that environment gsplat silently disables its CUDA backend ("No CUDA
# toolkit found") and the first rasterization raises
# `AttributeError: 'NoneType' object has no attribute 'CameraModelType'`.
#
# Install a PRE-BUILT wheel instead -- it ships precompiled kernels, so no
# nvcc is needed. Pick the wheel matching the env's torch + CUDA build from
# https://docs.gsplat.studio/whl/ . For Isaac Sim 4.5 (torch 2.5.1+cu118)
# the pt24cu118 wheel loads cleanly (the 2.4->2.5 ABI gap is a non-issue):
#
#   pip install --index-url https://docs.gsplat.studio/whl/pt24cu118 \
#       'gsplat==1.5.3+pt24cu118'
#
# (no pt25 wheels are published yet; pt24cu118 is the right match.)
# gsplat==1.5.3+pt24cu118
